225void detectar_caida(void* pvParameters) {
226 while (1) {
227 AnalogInputReadSingle(CH_X, &aceleracion_x);
228 AnalogInputReadSingle(CH_Y, &aceleracion_y);
229 AnalogInputReadSingle(CH_Z, &aceleracion_z);
230
231 float magnitud_aceleracion = sqrt(aceleracion_x * aceleracion_x + aceleracion_y * aceleracion_y + aceleracion_z * aceleracion_z);
232
233 if (magnitud_aceleracion > UMBRAL_CAIDA) {
234 enviar_notificacion_bluetooth("Caída detectada");
235 }
236 vTaskDelay(pdMS_TO_TICKS(10)); // a 100 Hz
237 }
238}
