154void mostrarTask()
155{
156 while (true)
157 {
158 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
159 if (on)
160 {
161 if (distancia < 10)
162 {
163 LedsOffAll();
164 }
165 else if ((distancia > 10) & (distancia < 20))
166 {
167 LedOn(LED_1);
168 LedOff(LED_2);
169 LedOff(LED_3);
170 }
171
172 else if ((distancia > 20) & (distancia < 30))
173 {
174 LedOn(LED_1);
175 LedOn(LED_2);
176 LedOff(LED_3);
177 }
178
179 else if (distancia > 30)
180 {
181 LedOn(LED_1);
182 LedOn(LED_2);
183 LedOn(LED_3);
184 }
185
186 if (!hold)
187 {
188 LcdItsE0803Write(distancia);
189 }
190 }
191
192 else
193 {
194 LcdItsE0803Off();
195 LedsOffAll();
196 }
197 }
