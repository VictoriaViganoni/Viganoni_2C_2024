141void mover_motor(int steps, gpio_t dirPin, gpio_t stepPin) {
142 GPIOState(dirPin, steps > 0 ? true : false);
143 printf("Moviendo motor en %s con %d pasos.\n", (dirPin == X_DIR_PIN ? "X" : "Y"), steps);
144 for (int i = 0; i < abs(steps); i++) {

179void traducir_y_troquelar(const char* texto) {
180 printf("Traduciendo y troquelando la palabra: %s\n", texto);
181 for (int i = 0; i < strlen(texto); i++) {
182 char caracter = texto[i];
183 if (caracter >= 'A' && caracter <= 'Z') {
184 int letra_idx = caracter - 'A';
185 const int* matriz = diccionario_braille[letra_idx];
186 printf("Troquelando letra: %c\n", caracter);
187
188
189 for (int col = 0; col < 2; col++) {
190 for (int row = 0; row < 3; row++) {
191 int punto_idx = row + col * 3;
192 if (matriz[punto_idx] == 1) {
193 troquelar();
194 }
195 if (row < 2) {
196 mover_motor(ESPACIO_ENTRE_PUNTOS, Y_DIR_PIN, Y_STEP_PIN);
197 }
198 }
199
200 mover_motor(-ESPACIO_ENTRE_PUNTOS * 2, Y_DIR_PIN, Y_STEP_PIN);
201 if (col == 0) {
202 mover_motor(ESPACIO_ENTRE_PUNTOS, X_DIR_PIN, X_STEP_PIN);
203 }
204 }
205 mover_motor(-ESPACIO_ENTRE_PUNTOS, X_DIR_PIN, X_STEP_PIN);
206 mover_motor(ESPACIO_ENTRE_CARACTERES, X_DIR_PIN, X_STEP_PIN);
207 }
208 }
209}
